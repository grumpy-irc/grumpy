PROJECT(grumpy)

option(QT6_BUILD "Build with Qt6" false)
SET(QT_USE_QTNETWORK TRUE)
SET(QT_USE_QTSCRIPT TRUE)
SET(CMAKE_BUILD_TYPE Release)
SET(CMAKE_AUTOMOC ON)

if(QT6_BUILD)
  find_package(Qt6Core REQUIRED)
  find_package(Qt6Widgets REQUIRED)
  find_package(Qt6Multimedia REQUIRED)
  find_package(Qt6Xml REQUIRED)
  find_package(Qt6Gui REQUIRED)
  find_package(Qt6Network REQUIRED)
  set(QT_INCLUDES ${Qt6Gui_INCLUDE_DIRS} ${Qt6Widgets_INCLUDE_DIRS} ${Qt6Network_INCLUDE_DIRS})
  include_directories(${QT_INCLUDES})
elseif(QT5_BUILD)
  find_package(Qt5Core REQUIRED)
  find_package(Qt5Widgets REQUIRED)
  find_package(Qt5Multimedia REQUIRED)
  find_package(Qt5Xml REQUIRED)
  find_package(Qt5Gui REQUIRED)
  find_package(Qt5Network REQUIRED)
  set(QT_INCLUDES ${Qt5Gui_INCLUDE_DIRS} ${Qt5Widgets_INCLUDE_DIRS} ${Qt5Network_INCLUDE_DIRS})
  include_directories(${QT_INCLUDES})
else()
  find_package(Qt4 REQUIRED)
endif()

file (GLOB ui "*.ui")
file (GLOB headers "*.h")
file (GLOB src "*.cpp")
file (GLOB resources "*.qrc")

if(QT6_BUILD)
    QT6_WRAP_CPP(grumpy_HEADERS_MOC ${grumpy_HEADERS})
    QT6_WRAP_UI(grumpy_FORMS_HEADERS ${ui})
    QT6_ADD_RESOURCES(grumpy_RESOURCES_RCC ${resources})
elseif(QT5_BUILD)
    QT5_WRAP_CPP(grumpy_HEADERS_MOC ${grumpy_HEADERS})
    QT5_WRAP_UI(grumpy_FORMS_HEADERS ${ui})
    QT5_ADD_RESOURCES(grumpy_RESOURCES_RCC ${resources})
else()
    QT4_WRAP_CPP(grumpy_HEADERS_MOC ${irc_HEADERS})
    QT4_WRAP_UI(grumpy_FORMS_HEADERS ${ui})
    QT4_ADD_RESOURCES(grumpy_RESOURCES_RCC ${resources})
endif()

if (NOT QT5_BUILD)
  INCLUDE(${QT_USE_FILE})
endif()

ADD_DEFINITIONS(${QT_DEFINITIONS})
ADD_DEFINITIONS( -DQT_USE_QSTRINGBUILDER )

ADD_EXECUTABLE(grumpy ${src} ${grumpy_HEADERS_MOC} ${headers} ${grumpy_RESOURCES_RCC} ${grumpy_FORMS_HEADERS})
if (QT6_BUILD)
    TARGET_LINK_LIBRARIES(grumpy Qt6::Core Qt6::Multimedia Qt6::Gui Qt6::Widgets Qt6::Xml)
elseif (QT5_BUILD)
    TARGET_LINK_LIBRARIES(grumpy Qt5::Core Qt5::Multimedia Qt5::Gui Qt5::Widgets Qt5::Xml)
endif()
TARGET_LINK_LIBRARIES(grumpy grumpy_lib grumpycore irc2htmlcode ${QT_LIBRARIES})
if (NOT WIN32)
    INSTALL(TARGETS grumpy RUNTIME DESTINATION bin)
endif()
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})
INCLUDE_DIRECTORIES("../../")
